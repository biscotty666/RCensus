tbl_uvsurv <- 
  trial |> 
  select(age, grade, death, ttdeath) |> 
  tbl_uvregression(
    method = coxph,
    y = Surv(ttdeath, death),
    exponentiate = T

  add_global_p()

tbl_mvsurv <- 
  coxph(
    Surv(ttdeath, death) ~ age + grade,

  tbl_regression(

stack for vertical

# Unadjusted model
t3 <- 
  coxph(Surv(ttdeath, death) ~ trt,
        data = trial) |> 
  tbl_regression(
    show_single_row = trt,         # Combine the three treatment types
    label = trt ~ "Drub B vs A",
    exponentiate = T
  )

# Adjusted model
t4 <- 
  coxph(Surv(ttdeath, death) ~
          trt + grade + stage + marker,
        data = trial) |> 
  tbl_regression(
    show_single_row = trt,
    label = trt ~ "Drug B vs A",
    exponentiate = T,
    include = "trt"
  )
tbl_stack(
  list(t3, t4),
  group_header = c("Unadjusted", "Adjusted")
)
